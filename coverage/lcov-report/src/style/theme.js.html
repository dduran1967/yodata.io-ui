export function scale(value?: number): string {
  switch (typeof value) {
    case "number":
      return `${value * 0.5}em`;
    case "string":
      return value;
    default:
      return "";
  }
}
 
export function padding(a: number, b?: number, c?: number, d?: number) {
  let padding = [scale(a) , scale(b) , scale(c) , scale(d)].join(' ').trim();
  return function (src: JSON) {
    return {...src, padding};
  };
}
