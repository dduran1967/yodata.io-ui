class ClientStore {
  constructor() {
    extendObservable(this, {
      index: [],
      items: []
    });
    this.add = this.add.bind(this);
    this.list = this.list.bind(this);
    this.load = this.load.bind(this);
    this.load();
  }
 
  load() {
    CLIENTS.forEach(this.add);
  }
 
  list() {
    return this.items;
  }
 
  get(id) {
    return this.index[id];
  }
 
  add(client) {
    this.index[client.id] = client;
    this.items.push(client);
  }
 
}
